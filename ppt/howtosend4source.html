getImageData: function(str){
                    var tmp = str.split(',')[0];
                    if (tmp.indexOf('data:image/png;base64') < 0 &&
                    tmp.indexOf('data:image/jpg;base64') < 0 &&
                    tmp.indexOf('data:image/jpeg;base64') < 0 &&
                    tmp.indexOf('data:image/bmp;base64') < 0 &&
                    tmp.indexOf('data:image/gif;base64') < 0) {
                        return;
                    }
                    return atob(str.split(',')[1]);
                }
